  // ============================================================================

  _match(expected) {

### 5.2 Generate Iterative Parser with CORRECTED Separator Logic

**Key correction:** State saved BEFORE matching separator, so restore undoes the match.

```coffee
_generateIterativeParser: (typeName, recursionInfo) ->
  {prefixRules, postfixRules, separator, elementNonterminal} = recursionInfo
  
  # Use first prefix rule as base case
  baseRule = prefixRules[0]
  baseNonterminal = baseRule.symbols[0]
  
  # Compute FIRST set of element nonterminal
  firstSet = Array.from(@types[elementNonterminal].firsts)
  firstTokenIds = firstSet.map((name) => 'SYM_' + name.toUpperCase().replace(/[^A-Z0-9]/g, '_'))
  
  # Get action from first postfix rule
  postfixRule = postfixRules[0]
  [captures, returnStmt] = @_compileActionForPRD(postfixRule.action, postfixRule.symbols)
  
  """
  parse#{typeName}() {
    // Base case (first prefix rule)
    $1 = this.parse#{baseNonterminal}();
    const elements = [$1];
    const separators = [];
    
    // Iterative loop with separator restoration (Fix #16)
    while (this.la && this.la.id === SYM_#{separator}) {
      const _saved = this._saveState();  // CRITICAL: Save BEFORE consuming
      
      const sep = this._match(SYM_#{separator});  // Consume separator
      
      // Check if element follows AFTER consuming separator
      if (!this.la || ![#{firstTokenIds.join(', ')}].includes(this.la.id)) {
        // No element follows - restore state (undoes the match)
        this._restoreState(_saved);
        break;
      }
      
      // Element follows - keep separator and parse element
      separators.push(sep);
      const elem = this.parse#{elementNonterminal}();
      elements.push(elem);
    }
    
    // Build result based on action pattern
    // Most actions are [...$1, $3] (discard separator)
    // Some need separator interleaved
    return elements;  // Action determines final structure
  }
  """
```

**Why this works:**
- `_saveState()` captures position BEFORE `_match()` advances it
- `_restoreState()` resets position, effectively "un-consuming" the separator
- Prevents orphaned separators in token stream

## Phase 6: Array Elision Architecture (5-7 hours)

### 6.1 Elision-First Ordering

**Solves Tests #1, 4, 6:** array destructuring skip, trailing comma, elision destructuring

**The problem:** Array has 3 rules all starting with `[`:
- `[ ]` - Empty array
- `[ Elisions ]` - Pure elisions like `[,,]`
- `[ ArgElisionList OptElisions ]` - Mixed like `[1,,2]`

**The bug in solar-old.rip:** Tried mixed pattern first, pure elisions last → `[,,1]` failed.

**The fix:** Order by specificity (empty → pure → mixed):

```coffee
_orderArrayAlternatives: (alternatives) ->
  # Scoring system for Array rules
  scoreRule = (rule) =>
    symbols = rule.symbols.join(' ')
    
    # Exact pattern matching
    return 10 if symbols is '[ ]'                     # Empty: [ ]
    return 9 if symbols is '[ Elisions ]'             # Pure: [,,]
    return 5 if symbols.includes('ArgElisionList')   # Mixed: [1,,2]
    return 0  # Default
  
  # Sort by score descending (higher = try first)
  alternatives.sort (a, b) => scoreRule(b.rule) - scoreRule(a.rule)
```

**Generic principle:** Detect by rule structure (symbol names), not hardcoded type checks.

### 8.1 Detect Conflicts from Parse Table

```coffee
_findConflicts: (typeName) ->
  conflicts = new Map  # triggerToken → [alternatives]
  
  # Scan all states for conflicts involving this nonterminal
  for state in @states
    # Look at all items in this state
    reductionsByLookahead = new Map
    
    for item from state.items
      if !item.nextSymbol and item.rule.type is typeName
        # Reduction item for our nonterminal
        for lookahead from item.lookaheads
          reductions = reductionsByLookahead.get(lookahead) or []
          reductions.push(item.rule)
          reductionsByLookahead.set(lookahead, reductions)
    
    # Check for conflicts
    for [lookahead, rules] from reductionsByLookahead
      if rules.length > 1 or state.hasShifts
        # Multiple reductions or shift/reduce conflict
        token = @tokenNames[lookahead] or lookahead
        conflicts.get(token) or conflicts.set(token, [])
        rules.forEach (rule) => conflicts.get(token).push({ rule, type: 'reduce' })
  
  conflicts
```

### 9.2 Generate Precedence Climbing Parser

_generateExpressionParser: ->
  # Expression has special handling for operators
  """
  parseExpression(minPrec = 0) {
    // Parse initial value
    let $1 = this.parseValue();
    
    // Precedence climbing loop
    while (this.la && OPERATOR_PRECEDENCE[this.la.id] !== undefined && 
           OPERATOR_PRECEDENCE[this.la.id] >= minPrec) {
      const prec = OPERATOR_PRECEDENCE[this.la.id];
      const assoc = OPERATOR_ASSOC[this.la.id] || 'left';
      
      // Match operator
      const op = this._match(this.la.id);
      
      // Parse right side with appropriate precedence
      const nextPrec = assoc === 'right' ? prec : prec + 1;
      const right = this.parseExpression(nextPrec);
      
      // Build binary operator node: [op, left, right]
      $1 = [op, $1, right];
    }
    
    return $1;
  }
  """

## Phase 10: Complete Action Compilation (6-8 hours)

### 10.1 Handle All Action Types

**Complete specification** (no truncation):

```coffee
_compileActionForPRD: (action, symbols) ->
  # Return: { captures: [...], return: 'return ...' }
  
  switch typeof action
    when 'string'
      if action is 'null'
        # GENERIC FIX #21: Single-terminal null action
        if symbols.length is 1 and !@types[symbols[0]]
          return { captures: [], return: 'return $1;' }
        else
          return { captures: [], return: 'return null;' }
      
      else if action is '""' or action is "''"
        # Empty string literal
        return { captures: [], return: 'return "";' }
      
      else if action.match(/^\[/)
        # S-expression array: ["def", 2, 4, 6]
        @_compileSExprAction(action, symbols)
      
      else if action.match(/\$\d/)
        # Complex JavaScript with positions: '$1 ? ["?call", $2] : [$2]'
        @_compileComplexAction(action, symbols)
      
      else
        # String literal: '"this"', '"null"'
        { captures: [], return: "return #{action};" }
    
    when 'number'
      # Single position reference: 2 means return $2
      pos = action
      capture = @_generateCapture(symbols[pos - 1], pos)
      { captures: [capture], return: "return $#{pos};" }
    
    when 'undefined'
      # Default action: return first symbol
      capture = @_generateCapture(symbols[0], 1)
      { captures: [capture], return: 'return $1;' }
    
    else
      # Unknown type
      throw new Error("Unknown action type: #{typeof action}")
```

### 10.2 S-Expression Action Compilation

```coffee
_compileSExprAction: (action, symbols) ->
  # Parse: '["def", 2, 4, 6]' or '[...1, 3]'
  # Extract position references
  
  positions = new Set
  
  # Match bare numbers (not in strings, not after $)
  action.replace /(?<!["\$\.])\b(\d+)\b/g, (_, n) =>
    positions.add(parseInt(n))
    return n
  
  # Sort positions for orderly capture generation
  sortedPositions = Array.from(positions).sort((a, b) => a - b)
  
  # Generate captures only for referenced positions
  captures = []
  for pos in sortedPositions when pos > 0 and pos <= symbols.length
    captures.push @_generateCapture(symbols[pos - 1], pos)
  
  # Replace position numbers with $N variables
  returnCode = action.replace /(?<!["\$\.])\b(\d+)\b/g, (_, n) -> "$#{n}"
  
  { captures, return: "return #{returnCode};" }

_generateCapture: (symbol, position) ->
  return '' if symbol is ''  # Empty symbol (ε)
  
  if @types[symbol]
    # Nonterminal - call parser
    "const $#{position} = this.parse#{symbol}();"
  else
    # Terminal - match token
    safeName = symbol.toUpperCase().replace(/[^A-Z0-9]/g, '_')
    "const $#{position} = this._match(SYM_#{safeName});"
```

### 10.3 Complex Action Compilation

```coffee
_compileComplexAction: (action, symbols) ->
  # Handle complex JavaScript actions with conditions
  # Example: '$1 ? ["?call", $2, ...$3] : [$2, ...$3]'
  # Example: 'Array.isArray($1) ? [...$1, $3] : [$1, $3]'
  
  # Find all $N references
  positions = new Set
  action.replace /\$(\d+)/g, (_, n) =>
    positions.add(parseInt(n))
    return ''
  
  # Generate captures for all referenced positions
  captures = []
  for pos in Array.from(positions).sort()
    captures.push @_generateCapture(symbols[pos - 1], pos)
  
  # Return action as-is (already has $N variables)
  { captures, return: "return #{action};" }
```
